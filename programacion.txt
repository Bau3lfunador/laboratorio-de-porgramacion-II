2.)
El diagrama de flujo muestra un programa que recibe tres notas, calcula su promedio y muestra el resultado
Este es el codigo para hacer eso: 
#include <iostream>
using namespace std;
int main() {
    float nota1, nota2, nota3, prom;
    cout << "ponga la primera nota: ";
    cin >> nota1;
    cout << "ponga la segunda nota: ";
    cin >> nota2;
    cout << "ponga la tercera nota: ";
    cin >> nota3;
    prom = (nota1 + nota2 + nota3) / 3;
    if (prom >= 7) {
        cout << "promocionado" << endl;
    } else if (prom >= 4) {
        cout << "Regular" << endl;
    } else {
        cout << "Reprobado" << endl;
    }
    return 0;
}

#include <iostream>
using namespace std;
int main() {
    int cantidad;
    float precioUnitario, total;
    cout << "ingrese el precio de una entrada: ";
    cin >> precioUnitario;
    cout << "ingrese la cantidad de entradas (maximo 4): ";
    cin >> cantidad;
    if (cantidad < 1 || cantidad > 4) {
        cout << "cantidad invalida solo se pueden comprar hasta 4 entradas." << endl;
    } else {
        total = precioUnitario * cantidad;
        if (cantidad == 2) {
            total *= 0.90;
        } else if (cantidad == 3) {
            total *= 0.85;
        } else if (cantidad == 4) {
            total *= 0.80;
        }
        cout << "el total a pagar es: $" << total << endl;
    }
    return 0;
}
